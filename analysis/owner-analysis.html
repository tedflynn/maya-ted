Total Lobsters by Year

lobster_abundance %>%
  group_by(SITE,YEAR) %>% 
  summarize(Total = sum(COUNT)) %>% 
  ggplot(aes(x = YEAR,
             y = Total,
             color = SITE)) +
  geom_line() +
  geom_point() +
  theme_bw() +
  labs(x = "Year",
       y = "Total Lobsters",
       color = "Site",
       title = "Lobster Abundance by Year and Site")

Compare Large and Small Salmon

lobster_size_lrg <- lobster_abundance %>% 
  filter(YEAR %in% c(2019, 2020, 2021)) %>% 
  mutate(SIZE_BIN = if_else(SIZE_MM >= 70, "large","small"))

lobster_size_lrg %>% 
  group_by(SITE, SIZE_BIN) %>% 
  summarize(TOTAL = sum(COUNT)) %>% 
  drop_na() %>% 
  ggplot(aes(x = SITE,
             y = TOTAL,
             fill = SIZE_BIN)) +
  geom_col(position = "dodge") +
  theme_bw() +
  labs(x = "Site Name",
       y = "Total Lobsters",
       fill = "Size Class",
       title = "Abundance of Lobsters by Size Class")
